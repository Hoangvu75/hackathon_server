Restful API request

GET user: "https://hackathon-server-project.herokuapp.com/users"

• Need no body



POST register: "https://hackathon-server-project.herokuapp.com/register"

• Body: { username: String, password: String }

• Example: { username: "admin123", password: "12345678" }



POST login: "https://hackathon-server-project.herokuapp.com/login"

• Body: { username: String, password: String }

• Example: { username: "admin123", password: "12345678" }



POST create user: "https://hackathon-server-project.herokuapp.com/create_user"

• Body: { name: String, age: Int, university: String, class: String }

• Example: { name: "Vũ Huy Hoàng", age: 19, university: "UIT", class: "PMCL2021.2" }



POST change password: "https://hackathon-server-project.herokuapp.com/change_password"

• Body: { username: String, password: String, newPassword: String, retypeNewPassword: String }

• Example: { username: "admin123", password: "12345678", newPassword: "87654321", retypeNewPassword: "87654321" }
